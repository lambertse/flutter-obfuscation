cmake_minimum_required(VERSION 3.18)
project(guard C)

# This project is intentionally NOT wired into the Flutter app's Gradle
# build (see Handover.md architecture: libguard.so is produced by a
# standalone post-processing step and injected into the already-built APK,
# not compiled as part of it). build_and_pack.sh invokes `cmake` directly,
# once per ABI, with the NDK's own toolchain file:
#
#   cmake -B build/<abi> -G Ninja \
#     -DCMAKE_TOOLCHAIN_FILE=$ANDROID_NDK_HOME/build/cmake/android.toolchain.cmake \
#     -DANDROID_ABI=<abi> -DANDROID_PLATFORM=android-<minSdk> \
#     -DCMAKE_BUILD_TYPE=Release \
#     packer/native/libguard
#   cmake --build build/<abi>
#
# One CMake configure = one ABI = one libguard.so; that's why this file has
# no per-ABI branching of its own (regions.h does the per-ABI branching via
# preprocessor, see region_table.h).

if(NOT ANDROID)
  message(FATAL_ERROR
    "This must be configured with the Android NDK's toolchain file "
    "(-DCMAKE_TOOLCHAIN_FILE=.../build/cmake/android.toolchain.cmake) and "
    "-DANDROID_ABI=<abi>. See tools/build_and_pack.sh.")
endif()

if(NOT EXISTS "${CMAKE_CURRENT_SOURCE_DIR}/src/regions.h")
  message(FATAL_ERROR
    "src/regions.h is missing. It is GENERATED by tools/encrypt_snapshot.py "
    "-- run the packer pipeline (tools/build_and_pack.sh) before building "
    "this CMake project directly.")
endif()

add_library(guard SHARED
  src/guard.c
  src/got_hook.c
  src/trampoline.c
  src/crypto.c
  src/anti_instr.c
  src/third_party/aes.c
)

target_include_directories(guard PRIVATE src)

# tiny-AES-c is vendored generically (all of ECB/CBC/CTR, all key sizes);
# we only need AES-256-CTR. Applied target-wide, NOT just to
# third_party/aes.c: crypto.c also #includes third_party/aes.h directly
# (for its AES_KEYLEN/AES_BLOCKLEN static_asserts against
# GUARD_AES_KEY_LEN/GUARD_NONCE_LEN), so it needs to see the same
# AES256=1 definition or aes.h silently falls back to its own default
# (AES128, 16-byte keys) for that translation unit and the static_assert
# in crypto.c fails to build. Caught by an actual compile, not review --
# see README "Known limitations".
target_compile_definitions(guard PRIVATE CBC=0 ECB=0 CTR=1 AES128=0 AES256=1)

find_library(log-lib log)
target_link_libraries(guard PRIVATE ${log-lib})

target_compile_options(guard PRIVATE
  -Wall -Wextra
  -fvisibility=hidden        # keep our internal symbols out of the dynamic table;
                              # guard_ctor runs via .init_array, not an exported symbol
  -ffunction-sections -fdata-sections
)
target_link_options(guard PRIVATE -Wl,--gc-sections)

# JNI_OnLoad is not used (see guard.c: key hand-off goes through an
# explicit Java_..._nativeSetKey export instead), so nothing else needs to
# be exported. Keep only what's required: the constructor is invoked by the
# dynamic linker directly, not looked up by name.
